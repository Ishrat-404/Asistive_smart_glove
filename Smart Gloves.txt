#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include "MAX30100_PulseOximeter.h"
#include <BluetoothSerial.h>

// --- DISPLAY SETTINGS ---
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

// --- SENSOR & BLUETOOTH SETTINGS ---
PulseOximeter pox;
BluetoothSerial SerialBT;

#define REPORTING_PERIOD_MS 1000
uint32_t tsLastReport = 0;

// --- FLEX SENSOR SETTINGS ---
#define FLEX_SENSOR_PIN 34  
#define FLEX_THRESHOLD 700  

// --- VITAL SIGN THRESHOLDS ---
#define SPO2_MIN_NORMAL 90
#define HR_MIN_NORMAL 40
#define HR_MAX_NORMAL 120

// --- VARIABLES ---
int hr = 0;
int spo2 = 0;
int patientStatus = 1; // 1 = OK, 0 = Alert
bool validReading = false;

// Callback (must be kept short)
void onBeatDetected() {
  Serial.println("Beat detected!");
}

void setup() {
  Serial.begin(115200);

  // 1. Initialize Bluetooth
  SerialBT.begin("ESP32_PulseOximeter"); 
  Serial.println("Bluetooth Started! Ready to pair...");

  // 2. Initialize OLED
  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println(F("SSD1306 allocation failed"));
    while (true); // Stop here if display fails
  }
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(0, 0);
  display.println("Initializing...");
  display.display();
  delay(1000);

  // 3. Initialize MAX30100
  Serial.print("Initializing Pulse Oximeter..");
  if (!pox.begin()) {
    Serial.println("FAILED");
    display.clearDisplay();
    display.setCursor(0, 10);
    display.println("Sensor Init Failed");
    display.println("Check Connections!");
    display.display();
    while (true); // Stop here if sensor fails
  } else {
    Serial.println("SUCCESS");
  }

  // Set LED current (adjust if sensor is too dim/bright)
  pox.setIRLedCurrent(MAX30100_LED_CURR_7_6MA);
  pox.setOnBeatDetectedCallback(onBeatDetected);

  // 4. Initialize Flex Sensor
  pinMode(FLEX_SENSOR_PIN, INPUT);

  // Show ready message
  display.clearDisplay();
  display.setCursor(0, 20);
  display.println("Place finger");
  display.println("on sensor...");
  display.display();
}

void loop() {
  // Make sure to call update as fast as possible
  pox.update();

  // Report and Send Data every 1 second
  if (millis() - tsLastReport > REPORTING_PERIOD_MS) {
    tsLastReport = millis();

    // -- A. READ VALUES (convert to int to remove decimals) --
    hr = (int)pox.getHeartRate();
    spo2 = (int)pox.getSpO2();
    int flexValue = analogRead(FLEX_SENSOR_PIN);

    // -- B. VALIDATE SENSOR READINGS --
    // Check if readings are within realistic human ranges
    validReading = (hr > 30 && hr < 220 && spo2 > 70 && spo2 <= 100);

    // -- C. LOGIC FOR PATIENT STATUS --
    if (!validReading) {
      // No valid reading from sensor - neutral status
      patientStatus = 1; // Don't trigger alert for no finger
      
      // -- SEND TO SERIAL MONITOR --
      Serial.println("No valid sensor reading - Place finger on sensor");
      
      // -- SEND TO BLUETOOTH --
      SerialBT.println("0,0,1"); // Status 2 = No Reading
      
      // -- UPDATE OLED DISPLAY --
      display.clearDisplay();
      display.setTextSize(1);
      display.setCursor(0, 10);
      display.println("Place finger");
      display.println("on sensor...");
      display.setCursor(0, 40);
      display.print("Flex: "); 
      display.println(flexValue);
      display.display();
      
    } else {
      // Valid reading - check patient status
      patientStatus = 1; // Default to OK

      // Condition 1: Flex sensor indicates discomfort
      if (flexValue < FLEX_THRESHOLD) {
        patientStatus = 0; // Not OK
        Serial.println("ALERT: Patient discomfort detected!");
      }
      
      // Condition 2: SpO2 too low
      if (spo2 < SPO2_MIN_NORMAL) { 
        patientStatus = 0;
        Serial.println("ALERT: Low SpO2 detected!");
      }

      // Condition 3: Abnormal heart rate
      if (hr < HR_MIN_NORMAL || hr > HR_MAX_NORMAL) {
        patientStatus = 0;
        Serial.println("ALERT: Abnormal heart rate detected!");
      }

      // -- D. SEND TO SERIAL MONITOR (PC) --
      Serial.print("HR: "); Serial.print(hr);
      Serial.print(" bpm / SpO2: "); Serial.print(spo2);
      Serial.print("% / Flex: "); Serial.print(flexValue);
      Serial.print(" / Status: "); 
      if (patientStatus == 1) {
        Serial.println("OK");
      } else {
        Serial.println("ALERT");
      }

      // -- E. SEND TO BLUETOOTH (MOBILE) --
      // Format: "HR,SpO2,Status" (e.g., "75,98,1")
      // Status: 0 = Alert, 1 = OK, 2 = No Reading
      String dataPacket = String(hr) + "," + String(spo2) + "," + String(patientStatus);
      SerialBT.println(dataPacket);

      // -- F. UPDATE OLED DISPLAY --
      display.clearDisplay();
      display.setTextSize(1);
      
      // Heart Rate
      display.setCursor(0, 0);
      display.print("HR: "); 
      display.print(hr); 
      display.println(" bpm");

      // SpO2
      display.setCursor(0, 12);
      display.print("SpO2: "); 
      display.print(spo2); 
      display.println(" %");

      // Flex Sensor
      display.setCursor(0, 24);
      display.print("Flex: "); 
      display.println(flexValue);

      // Status Message
      display.setCursor(0, 40);
      if (patientStatus == 1) {
        display.println("Status: OK");
        display.setTextSize(2);
        display.setCursor(0, 52);
        display.println("NORMAL");
      } else {
        display.setTextSize(1);
        display.println("Status: ALERT!");
        display.setTextSize(2);
        display.setCursor(0, 52);
        display.println("CHECK!");
      }
      
      display.display();
    }
  }
}
